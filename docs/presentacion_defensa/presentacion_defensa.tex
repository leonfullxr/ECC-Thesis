\section{Fundamentos: curvas elípticas}

\begin{frame}{¿Qué es una curva elíptica?}
  \footnotesize
  El conjunto de soluciones de una ecuación cúbica sobre un cuerpo finito, más un
  \alert{punto del infinito} $\mathcal{O}$.
  \begin{columns}[t]
    \begin{column}{0.5\textwidth}
      \centering
      \textbf{Cuerpos primos $\mathbb{F}_p$}\; $\boxed{\,y^2 = x^3 + ax + b\,}$\\[0.4ex]
      \footnotesize \alert{secp256k1} (Bitcoin), \alert{NIST P-256}, \alert{NIST P-384}
    \end{column}
    \begin{column}{0.5\textwidth}
      \centering
      \textbf{Cuerpos binarios $\mathbb{F}_{2^m}$}\;
      $\boxed{\,y^2 + xy = x^3 + ax^2 + b\,}$\\[0.4ex]
      \footnotesize 5 curvas \alert{SEC 2}: 3 Koblitz (\texttt{sect*k1}) $+$ 2 aleatorias
      (\texttt{sect*r1})
    \end{column}
  \end{columns}
  \begin{center}
    \figordemo[width=0.80\textwidth]{imagenes/ec_shape.png}
  \end{center}
\end{frame}

\begin{frame}{La ley de grupo: cómo se ``suman'' puntos}
  \begin{columns}[c]
    \begin{column}{0.42\textwidth}
      Los puntos forman un \alert{grupo abeliano} $(E(K),+)$:
      \begin{itemize}
        \item \textbf{Suma $P+Q$:} la recta por $P$ y $Q$ corta la curva en un tercer
              punto; su \alert{reflexión} es $P+Q$.
        \item \textbf{Duplicación $2P$:} igual, con la \alert{tangente} en $P$.
        \item \textbf{Neutro:} $\mathcal{O}$ (rectas verticales).
      \end{itemize}
      \vspace{0.5ex}
      \footnotesize Son las \alert{operaciones atómicas}: la \alert{multiplicación
      escalar} $kP$ es repetirlas (\emph{double-and-add}).
\end{column}
    \begin{column}{0.58\textwidth}
      \centering
      \figordemo[width=\textwidth]{imagenes/ec_addition.png}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{El grupo $(E(K),+)$: estructura y orden}
  \small
  \begin{columns}[t]
    \begin{column}{0.5\textwidth}
      \textbf{Grupo abeliano.} Para todo $P,Q,R$:
      \begin{itemize}\small
        \item Clausura: $P+Q\in E(K)$
        \item Asociatividad: $(P+Q)+R=P+(Q+R)$
        \item Neutro: $P+\mathcal{O}=P$
        \item Inverso: $P+(-P)=\mathcal{O}$
        \item Conmutatividad: $P+Q=Q+P$
      \end{itemize}
    \end{column}
    \begin{column}{0.5\textwidth}
      \textbf{Orden} $\#E(\mathbb{F}_q)$ (número de puntos):
      \begin{block}{Teorema de Hasse}
        $\bigl|\,\#E(\mathbb{F}_q) - (q+1)\,\bigr| \le 2\sqrt{q}$
      \end{block}
      En cripto: subgrupo cíclico de \alert{orden primo $n$}, con $\#E = h\cdot n$ (cofactor
      $h$ pequeño). La dificultad del ECDLP es $\sim\!\sqrt{n}$.
    \end{column}
  \end{columns}
  \note{Cinco axiomas de grupo abeliano; asociatividad la dificil. Orden = numero de puntos.
  Hasse: cerca de q+1. Estructura Z/n1 + Z/n2. Cripto: subgrupo de orden primo n, cofactor h
  pequeno, ECDLP ~ sqrt(n).}
\end{frame}

\begin{frame}{Suma y duplicación: coordenadas afines}
  \small
  \setlength{\fboxsep}{2pt}
  \setlength{\parskip}{0pt}
  \begin{columns}[t]
    \begin{column}{0.5\textwidth}
      {\color{UGRsoft}\textbf{Cuerpo primo}}\; $y^2=x^3+ax+b$
      \vspace{0.6ex}

      \emph{Suma} $(P\neq Q)$:\\[0.5ex]
      $\boxed{x_3=\frac{(y_2-y_1)^2}{(x_2-x_1)^2}-x_1-x_2}$\\[0.5ex]
      $\boxed{y_3=\frac{(y_2-y_1)(x_1-x_3)}{x_2-x_1}-y_1}$
      \vspace{0.8ex}

      \emph{Duplicación} $(2P)$:\\[0.5ex]
      $\boxed{x_3=\frac{(3x_1^2+a)^2}{4y_1^2}-2x_1}$\\[0.5ex]
      $\boxed{y_3=\frac{(3x_1^2+a)(x_1-x_3)}{2y_1}-y_1}$
    \end{column}
    \begin{column}{0.5\textwidth}
      {\color{UGRsoft}\textbf{Cuerpo binario}}\; $y^2+xy=x^3+ax^2+b$
      \vspace{0.6ex}

      \emph{Suma} $(P\neq Q)$:\\[0.5ex]
      $\boxed{x_3=\frac{(y_1+y_2)^2}{(x_1+x_2)^2}+\frac{y_1+y_2}{x_1+x_2}+x_1+x_2+a}$\\[0.5ex]
      $\boxed{y_3=\frac{(y_1+y_2)(x_1+x_3)}{x_1+x_2}+x_3+y_1}$
      \vspace{0.8ex}

      \emph{Duplicación} $(2P)$:\\[0.5ex]
      $\boxed{x_3=\frac{(x_1^2+y_1)^2}{x_1^2}+\frac{x_1^2+y_1}{x_1}+a}$\\[0.5ex]
      $\boxed{y_3=x_1^2+\Bigl(\frac{x_1^2+y_1}{x_1}+1\Bigr)x_3}$
    \end{column}
  \end{columns}
  \vspace{1ex}
  Cada operación necesita una \alert{división} (inversión modular) en el cuerpo.
  \note{Formulas finales expandidas, sin nombrar la pendiente lambda como variable aparte (esta
  ya sustituida dentro de x3, y3). Afin primo: resta/division al cuadrado; afin binario: sumas
  (char 2), division. Cada suma o duplicacion afin necesita una inversion modular, la operacion
  mas cara del cuerpo; eso motiva las Jacobianas de la siguiente diapositiva.}
\end{frame}

\begin{frame}{Suma y duplicación: coordenadas Jacobianas (cuerpo primo)}
  \small
  \setlength{\fboxsep}{2pt}
  \setlength{\parskip}{0pt}
  Punto como clase $(X{:}Y{:}Z)$ con $x=\dfrac{X}{Z^2}$, $y=\dfrac{Y}{Z^3}$:
  la $Z$ absorbe el denominador y \alert{no se invierte en cada paso}.
  \vspace{1ex}
  \begin{columns}[t]
    \begin{column}{0.5\textwidth}
      \emph{Suma} $(P\neq Q)$:\\[0.4ex]
      $U_1{=}X_1Z_2^2$, \quad $U_2{=}X_2Z_1^2$,\\
      $S_1{=}Y_1Z_2^3$, \quad $S_2{=}Y_2Z_1^3$,\\
      $H{=}U_2{-}U_1$, \quad $R{=}S_2{-}S_1$:\\[0.8ex]
      $\boxed{X_3=R^2-H^3-2U_1H^2}$\\[0.5ex]
      $\boxed{Y_3=R(U_1H^2-X_3)-S_1H^3}$\\[0.5ex]
      $\boxed{Z_3=HZ_1Z_2}$
    \end{column}
    \begin{column}{0.5\textwidth}
      \emph{Duplicación} $(2P)$:\\[0.4ex]
      $A{=}Y_1^2$, \quad $B{=}4X_1A$,\\
      $C{=}8A^2$, \quad $D{=}3X_1^2+aZ_1^4$:\\[0.8ex]
      $\boxed{X_3=D^2-2B}$\\[0.5ex]
      $\boxed{Y_3=D(B-X_3)-C}$\\[0.5ex]
      $\boxed{Z_3=2Y_1Z_1}$
    \end{column}
  \end{columns}
  \vspace{1ex}
  Suma: $12M+4S$ \quad · \quad Duplicación: $4M+4S$ \quad · \quad \alert{Sin inversiones}
  \note{Jacobiana primo: punto es clase (X:Y:Z) con x=X/Z^2, y=Y/Z^3 (Z absorbe el denominador,
  no se invierte en cada paso); suma con auxiliares U1,U2,S1,S2,H,R (12M+4S); duplicacion con
  A,B,C,D (4M+4S); ninguna necesita inversion. El porque del ahorro se explica en la siguiente
  diapositiva.}
\end{frame}

\begin{frame}{¿Por qué las Jacobianas son más eficientes?}
  \small
  \begin{columns}[c]
    \begin{column}{0.52\textwidth}
      \begin{itemize}\small
        \item La \alert{inversión modular} es la operación más cara del cuerpo.
        \item Modelo clásico: $I \approx 80\text{--}100\,M$ (multiplicaciones).
        \item Recuento de una \emph{duplicación}:
          \begin{itemize}\footnotesize
            \item Afín: $1I + 2M + 2S$
            \item Jacobiana: $4M + 6S$\; (\alert{sin $I$})
          \end{itemize}
        \item Si $I\gg M$, ganan las Jacobianas: \alert{una sola inversión al final}.
      \end{itemize}
    \end{column}
    \begin{column}{0.48\textwidth}
      \begin{alertblock}{Conexión con los benchmarks}
        La teoría predice $\sim$8$\times$; nosotros medimos $\sim$1.8$\times$.\\[0.6ex]
        NTL/GMP hacen la inversión tan rápida que el cociente real $I/M$ es mucho menor que
        $80$--$100$, y el ahorro se reduce.
      \end{alertblock}
    \end{column}
  \end{columns}
  \note{Inversion cara, I~80-100 M. Duplicacion afin 1I+2M+2S vs Jacobiana 4M+6S sin I. Ganan si
  I>>M, una inversion al final. Pero GMP hace I barata: I/M real << 80-100, por eso medimos
  1.8x y no 8x. Esto explica el Eje 2.}
\end{frame}

\begin{frame}{Multiplicación escalar y el problema difícil}
  \textbf{Multiplicación escalar:} $\;kP = \underbrace{P + P + \dots + P}_{k\ \text{veces}}$,
  calculada con \alert{double-and-add} en $O(\log k)$ pasos.
  \vfill
  \begin{columns}[c]
    \begin{column}{0.5\textwidth}
      \begin{exampleblock}{Fácil}
        Dados $k$ y $P$, calcular $Q = kP$.
      \end{exampleblock}
    \end{column}
    \begin{column}{0.5\textwidth}
      \begin{alertblock}{Inviable (ECDLP)}
        Dados $P$ y $Q=kP$, hallar $k$.
      \end{alertblock}
    \end{column}
  \end{columns}
  \vfill
  Esta asimetría es el \alert{Problema del Logaritmo Discreto en Curvas Elípticas (ECDLP)}.
  Sobre él descansan \textbf{ECDH} y \textbf{ECDSA}: $k$ es la clave privada, $Q=kP$ la
  pública.
  \note{kP con double-and-add O(log k). Facil k->Q, inviable Q->k = ECDLP. k privada, Q=kP
  publica. ECDH y ECDSA dependen de su dificultad.}
\end{frame}

\begin{frame}{Double-and-add: cómo se calcula $kP$}
  \small
  Se recorren los \alert{bits de $k$} de izquierda a derecha; en cada bit se \alert{duplica}
  ($\times2$) y, si el bit vale $1$, se \alert{suma $P$}.\;
  Ejemplo $k=13=(1101)_2$:
  \vspace{0.3em}

  \centering
  \resizebox{0.86\textwidth}{!}{%
  \begin{tikzpicture}[font=\footnotesize,
      bit/.style={draw, minimum size=0.72cm, font=\bfseries, inner sep=0pt},
      acc/.style={draw, rounded corners, fill=UGRsoft!12, minimum height=0.8cm,
                  minimum width=1.05cm},
      dbl/.style={-{Stealth}, very thick, UGRsoft},
      add/.style={-{Stealth}, very thick, UGRaccent}]

    % --- vector de bits ---
    \node (klbl) {$k=$};
    \node[bit, right=1.5mm of klbl] (kb3) {1};
    \node[bit, right=0mm of kb3] (kb2) {1};
    \node[bit, right=0mm of kb2] (kb1) {0};
    \node[bit, right=0mm of kb1] (kb0) {1};
    \node[font=\scriptsize, above=0pt of kb3] {MSB};
    \node[font=\scriptsize, above=0pt of kb0] {LSB};

  {\footnotesize \alert{$13P$ en 5 operaciones} (\textcolor{UGRsoft}{\textbf{3 duplicaciones}}
  $+$ \textcolor{UGRaccent}{\textbf{2 sumas}}), no 12 sumas: en general $\sim\!\log_2 k$
  pasos $\Rightarrow O(\log k)$.}
  \note{Double-and-add con k=13=1101. Se leen los bits MSB->LSB: primer 1 inicializa R=P;
  cada bit siguiente duplica y si es 1 suma P. Traza: P, 2P, 3P, 6P, 12P, 13P. Resultado en 5
  operaciones (3 dobles + 2 sumas) en lugar de 12 sumas. Coste O(log k), lo que hace kP
  eficiente (y el ECDLP sigue siendo dificil en sentido inverso).}
\end{frame}

\begin{frame}{Atacar el ECDLP: el algoritmo rho de Pollard}
  \begin{columns}[c]
    \begin{column}{0.44\textwidth}
      \footnotesize
      Mejor ataque \alert{genérico} conocido:
      \begin{itemize}\footnotesize
        \item Paseo \alert{pseudoaleatorio} sobre el grupo de puntos.
        \item Por la \alert{paradoja del cumpleaños}, aparece pronto una \alert{colisión}:
              el camino entra en un ciclo (forma de $\rho$: cola $+$ ciclo).
        \item Detección de ciclo con poca memoria (Floyd: tortuga/liebre). De la colisión se
              despeja $k$.
      \end{itemize}
      \vspace{0.5ex}
      \begin{block}{Coste}
        $O(\sqrt{n})$ en tiempo, $O(1)$ en memoria $\Rightarrow$ \alert{exponencial} en bits.
      \end{block}
    \end{column}
    \begin{column}{0.56\textwidth}
      \centering
      \figordemo[width=\textwidth]{imagenes/attack_pollard_rho.png}
    \end{column}
  \end{columns}
  \note{Rho de Pollard: paseo pseudoaleatorio, colision por cumpleanos, forma de rho, Floyd,
  se despeja k. O(sqrt n) tiempo, O(1) memoria. NO confundir con rho para factorizacion.}
\end{frame}

\section{Protocolos: ECDH, ECDSA y RSA}

\begin{frame}{Parámetros del dominio}
  \small
  Para instanciar ECC sobre $\mathbb{F}_p$, todas las partes acuerdan la séxtupla
  \[ T = (p,\; a,\; b,\; G,\; n,\; h) \]
  \begin{columns}[t]
    \begin{column}{0.5\textwidth}
      \begin{itemize}\footnotesize
        \item $p$: primo que define $\mathbb{F}_p$.
        \item $a,b$: coeficientes de $y^2=x^3+ax+b$.
        \item $G=(G_x,G_y)$: punto generador.
        \item $n$: orden de $G$ (menor con $nG=\mathcal{O}$).
        \item $h = \#E(\mathbb{F}_p)/n$: cofactor.
      \end{itemize}
    \end{column}
    \begin{column}{0.5\textwidth}
      \textbf{Requisitos de seguridad}
      \begin{itemize}\footnotesize
        \item $n$ primo (evita Pohlig--Hellman).
        \item $h$ pequeño, ideal $1$ (evita curva inválida).
        \item No anómala: $\#E(\mathbb{F}_p)\neq p$.
        \item Grado de inmersión alto (evita MOV).
        \item $\Delta = -16(4a^3+27b^2)\neq 0$.
      \end{itemize}
    \end{column}
  \end{columns}
  \vspace{0.5ex}
  \footnotesize En el TFG usamos curvas estándar \alert{NIST/SECG}, ya auditadas por la
  comunidad criptográfica.
  \note{Sextupla T=(p,a,b,G,n,h). p primo, a,b coeficientes, G generador, n orden de G, h
  cofactor. Requisitos: n primo, h pequeno, no anomala, inmersion alta, discriminante !=0.
  Usamos curvas estandar NIST/SECG.}
\end{frame}

\begin{frame}{Intercambio de claves: ECDH paso a paso}
  \centering
  \begin{tikzpicture}[font=\scriptsize]
    \node[hdr] (A) at (0,0) {Alice};
    \node[hdr] (B) at (9,0) {Bob};
    \draw[dashed, gray!60] (A) -- (0,-6.5);
    \draw[dashed, gray!60] (B) -- (9,-6.5);

    \node[sh, text width=6.2cm] (T) at (4.5,-0.95)
      {\textbf{0.} Acuerdan el dominio $T=(p,a,b,G,n,h)$ \;($G$ es público)};

    \node[sec, text width=3.2cm, anchor=north] (ka) at (0,-1.9)
      {\textbf{1.} genera su clave\\ $d_A \in_R [1,n{-}1]$ \;\textcolor{UGRgreen}{(privada)}\\ $Q_A = d_A\,G$ \;(pública)};
    \node[sec, text width=3.2cm, anchor=north] (kb) at (9,-1.9)
      {\textbf{1.} genera su clave\\ $d_B \in_R [1,n{-}1]$ \;\textcolor{UGRgreen}{(privada)}\\ $Q_B = d_B\,G$ \;(pública)};

    \draw[flow] (1.7,-4.05) -- node[above]{$Q_A$} (7.3,-4.05);
    \draw[flow] (7.3,-4.55) -- node[below]{$Q_B$} (1.7,-4.55);
    \node[lbl] at (4.5,-4.3) {2. canal público};

    \node[pub, text width=2.7cm, anchor=north] (sa) at (0,-5.0)
      {\textbf{3.} $S = d_A\,Q_B$};
    \node[pub, text width=2.7cm, anchor=north] (sb) at (9,-5.0)
      {\textbf{3.} $S = d_B\,Q_A$};

    \node[sh, text width=7.6cm, anchor=north] (res) at (4.5,-6.0)
      {\textbf{4.} $S = d_A Q_B = d_B Q_A = d_A d_B\,G$ \;(el \alert{mismo} punto)\\[2pt]
       $\Rightarrow$ clave de sesión $=\mathrm{KDF}(x_S)$};
  \end{tikzpicture}

  \vspace{0.3em}
  {\scriptsize \textcolor{UGRgreen}{verde = secreto} \quad azul = público \quad
   $\bullet$ un intruso ve $G,Q_A,Q_B$, pero hallar $d_A,d_B$ es el \alert{ECDLP}.}
  \note{Alice y Bob: privado d, publico Q=dG. Intercambian Q_A, Q_B por canal publico. Cada
  uno calcula S = d_A Q_B = d_B Q_A = d_A d_B G. Clave de sesion = KDF(x_S). El intruso ve los
  Q pero no los d (ECDLP). ECDH da confidencialidad; en efimero (ECDHE), secreto perfecto
  hacia adelante. NO autentica por si solo: hace falta firma/certificado contra MITM.}
\end{frame}

\begin{frame}{Firma digital: ECDSA paso a paso}
  \centering
  \begin{tikzpicture}[font=\scriptsize]
    \node[hdr, anchor=north] (Sh) at (0,0.1) {Firmante (Alice): privada $d$};
    \node[sec, text width=5.0cm, anchor=north] (sign) at (0,-0.4)
      {\textbf{1.} $e = H(m)$\\
       \textbf{2.} nonce $k \in_R [1,n{-}1]$ \;\textcolor{UGRgreen}{(secreto, único)}\\
       \textbf{3.} $R = kG = (x_R,y_R)$,\; $r = x_R \bmod n$\\
       \textbf{4.} $s = k^{-1}(e + d\,r) \bmod n$\\[2pt]
       $\Rightarrow$ firma $\boxed{(r,s)}$};

    \node[hdr, anchor=north] (Vh) at (8.8,0.1) {Verificador (Bob): pública $Q=dG$};
    \node[pub, text width=5.0cm, anchor=north] (ver) at (6.4,-0.4)
      {\textbf{1.} $e = H(m)$,\; $w = s^{-1} \bmod n$\\
       \textbf{2.} $u_1 = e\,w$,\; $u_2 = r\,w \pmod n$\\
       \textbf{3.} $R' = u_1 G + u_2 Q$\\
       \textbf{4.} aceptar si $x_{R'} \bmod n = r$};

    \draw[flow] (sign.east) -- node[above, font=\scriptsize]{$m,\,(r,s)$} (ver.west);

    \node[sh, text width=6.4cm, anchor=north] (ok) at (4.4,-3.6)
      {\textbf{Por qué cuadra:} $k = u_1 + u_2\,d \Rightarrow R' = (u_1+u_2 d)G = kG = R$};
  \end{tikzpicture}

  \vspace{0.2em}
  \begin{alertblock}{El nonce $k$ es crítico}
    \footnotesize Reutilizarlo o predecirlo $\Rightarrow$ se despeja la clave privada $d$
    (casos PlayStation 3 y Bitcoin). Solución: nonce determinista (RFC 6979).
  \end{alertblock}
  \note{Firma: e=H(m), nonce k, R=kG, r=x_R mod n, s=k^-1(e+dr). Firma (r,s). Verificacion:
  w=s^-1, u1=ew, u2=rw, R'=u1 G+u2 Q, aceptar si x_R' mod n = r. Correccion: k=u1+u2 d.
  Nonce k critico: reuso -> se despeja d (PS3, Bitcoin). RFC 6979 determinista.}
\end{frame}

\begin{frame}{El contraste: RSA paso a paso}
  \centering
  \begin{tikzpicture}[font=\scriptsize]
    \node[hdr] (B) at (0,0) {Bob (dueño de las claves)};
    \node[hdr] (A) at (9,0) {Alice (quiere enviar $m$)};
    \draw[dashed, gray!60] (0,-0.3) -- (0,-5.7);
    \draw[dashed, gray!60] (9,-0.3) -- (9,-5.7);

    \node[sec, text width=4.2cm, anchor=north] (kg) at (0,-0.55)
      {\textbf{1.} genera sus claves\\
       primos $p,q$ \;\textcolor{UGRgreen}{(secretos)}\\
       $n = p\,q$,\; $\varphi(n)=(p{-}1)(q{-}1)$\\
       $e = 65537$ \;(público)\\
       $d = e^{-1} \bmod \varphi(n)$ \;\textcolor{UGRgreen}{(privada)}};

    \draw[flow] (2.3,-2.5) -- node[above]{clave pública $(n,e)$} (9,-2.5);

    \node[pub, text width=3.4cm, anchor=north] (enc) at (9,-2.9)
      {\textbf{2.} cifra con $(n,e)$:\\ $c = m^{e} \bmod n$};

    \draw[flow] (9,-4.1) -- node[below]{texto cifrado $c$} (2.3,-4.1);

    \node[pub, text width=3.9cm, anchor=north] (dec) at (0,-4.5)
      {\textbf{3.} descifra con $d$:\\ $m = c^{d} \bmod n$ \;(acelera con CRT)};
  \end{tikzpicture}

  \vspace{0.3em}
  {\scriptsize \textcolor{UGRgreen}{verde = secreto}, azul = público.\;
   Romper RSA $=$ \alert{factorizar $n$} (subexponencial) $\Rightarrow$ claves grandes.}
  \note{RSA: Bob genera p,q secretos; n=pq, phi=(p-1)(q-1); e=65537 publico; d=e^-1 mod phi
  privada. Publica (n,e). Alice cifra c=m^e mod n. Bob descifra m=c^d mod n (CRT ~3.5x).
  Seguridad: factorizar n es subexponencial, por eso claves grandes. Contraste directo con ECC:
  misma seguridad con claves mucho menores (Eje 1). La firma RSA es analoga (m^d / m^e).}
\end{frame}

\begin{frame}{Elección del \emph{stack}: C++17 + NTL/GMP}
  \small
  \begin{columns}[t]
    \begin{column}{0.54\textwidth}
      \textbf{Lo elegido}
      \begin{itemize}\footnotesize
        \setlength{\itemsep}{2pt}
        \item \alert{C++17}: control total del camino caliente; sin recolector de basura
              ni intérprete que contaminen los tiempos.
        \item \alert{NTL 11.5.1 sobre GMP 6.3.0}: la referencia en aritmética de precisión
              arbitraria (\texttt{ZZ\_p} para $\mathbb{F}_p$, \texttt{GF2E} para
              $\mathbb{F}_{2^m}$).
        \item \alert{SHA-256 propio} (FIPS 180-4): el hash de ECDSA sin dependencias
              externas que alteren las medidas.
        \item OpenSSL 3.0.2: \textbf{solo} como \emph{baseline} externo de contraste.
      \end{itemize}
    \end{column}
    \begin{column}{0.46\textwidth}
      \textbf{Lo descartado (y por qué)}
      \begin{itemize}\footnotesize
        \setlength{\itemsep}{2pt}
        \item \textbf{Python / Sage}: esconden justo la aritmética que se quiere estudiar
              y medir.
        \item \textbf{Rust}: alternativa moderna razonable, pero con un ecosistema de
              multiprecisión menos maduro que NTL/GMP.
        \item \textbf{Construir sobre OpenSSL}: mediríamos su ensamblador a mano, no los
              algoritmos.
      \end{itemize}
    \end{column}
  \end{columns}
  \note{C++17: control del camino caliente, sin GC ni interprete. NTL sobre GMP: estandar de
  oro en multiprecision; ZZ_p para Fp (delega en ensamblador GMP), GF2E para F2m. SHA-256
  propio por integridad academica y para que ninguna dependencia altere los tiempos. OpenSSL
  solo baseline. Descartes: Python/Sage esconden la aritmetica (justo lo que se estudia);
  Rust razonable pero ecosistema bignum menos maduro; construir sobre OpenSSL mediria su
  ensamblador, no los algoritmos.}
\end{frame}

\section{Conclusiones y trabajo futuro}

\begin{frame}{Conclusiones: ECC frente a RSA}
  \begin{columns}[t]
    \begin{column}{0.5\textwidth}
      \begin{block}{ECC — cuándo elegirla}
        \footnotesize
        \textbf{Pros:} claves pequeñas (256\,b $\approx$ RSA-3072); generación de claves y
        \emph{firma} rápidas; menos ancho de banda, memoria y energía.\\[0.5ex]
        \textbf{Ideal para:} TLS 1.3 efímero, móvil/IoT, \emph{passkeys}, Signal/WhatsApp,
        Bitcoin/Ethereum (secp256k1).
      \end{block}
    \end{column}
    \begin{column}{0.5\textwidth}
      \begin{block}{RSA — cuándo sigue ganando}
        \footnotesize
        \textbf{Pros:} \emph{verificación} muy rápida ($e=65537$); esquema simple y ubicuo;
        enorme base instalada y compatibilidad.\\[0.5ex]
        \textbf{Ideal para:} PKI/certificados heredados, \emph{tokens}/JWT verificados muchas
        veces, sistemas legados de larga vida.
      \end{block}
    \end{column}
  \end{columns}
  \vfill
  \footnotesize\centering
  \alert{No hay ganador absoluto:} la ventaja de ECC en tamaño de clave es \textbf{estructural}
  (ECDLP exponencial) y \textbf{crece} con la seguridad; RSA conserva la ventaja en
  \textbf{verificación}. \\[0.3ex]
  \alert{El valor del trabajo: separar el álgebra de la ingeniería.}
  \note{ECC: claves pequenas, keygen y firma rapidas, menos ancho de banda/energia; ideal TLS
  1.3, movil/IoT, passkeys, Signal, Bitcoin. RSA: verificacion muy rapida (e=65537), simple,
  ubicuo, compatibilidad; ideal PKI heredada, JWT verificados muchas veces, legados. No hay
  ganador absoluto: ventaja de ECC estructural y crece; RSA gana en verificacion. Separar
  algebra de ingenieria.}
\end{frame}

\begin{frame}{ECC hoy en día: ¿dónde se usa?}
  \footnotesize
  \begin{itemize}\footnotesize
    \item \textbf{Web (TLS 1.3):} $\sim$97--98\% de los handshakes usan ECDHE; \alert{X25519}
      por defecto en navegadores.
    \item \textbf{Passkeys / FIDO2:} firmadas con \alert{ECDSA P-256}.
    \item \textbf{Mensajería y redes:} Signal, WhatsApp, SSH, WireGuard $\to$
      \alert{Curve25519 / Ed25519}.
    \item \textbf{Criptomonedas:} Bitcoin y Ethereum $\to$ \alert{secp256k1} (la curva de
      Koblitz del TFG).
    \item \textbf{Documentos:} pasaportes electrónicos (ICAO Doc 9303).
  \end{itemize}
  \vspace{0.5ex}
  \begin{alertblock}{La transición post-cuántica es \emph{híbrida}, no una retirada}
    \scriptsize \texttt{X25519MLKEM768} (X25519 $+$ ML-KEM-768) ya cubre $>$40\% del tráfico
    HTTPS humano en Cloudflare, por defecto en Chrome, Firefox, OpenSSL y Apple. X25519 sigue
    como \alert{salvaguarda clásica}.
  \end{alertblock}
  \note{TLS 97-98 por ciento ECDHE, X25519 default. Passkeys ECDSA P-256. Signal/WhatsApp/SSH/WireGuard
  Curve25519/Ed25519. Bitcoin/Ethereum secp256k1. Pasaportes ICAO. Post-cuantico HIBRIDO:
  X25519MLKEM768 >40 por ciento trafico Cloudflare, default en navegadores. ECC no se retira.}
\end{frame}

\begin{frame}[noframenumbering]{Respaldo - de dónde sale el $\sim$8$\times$ teórico (y el
    1{,}8$\times$ real)}
  \footnotesize
  Conteo para $kP$ de 256 bits ($\approx$256 duplicaciones $+$ 128 sumas), con
  $S\approx0{,}8M$:
  \begin{itemize}\footnotesize
    \item \textbf{Afín} ($1I+2M+2S$ por paso): $\approx 384\,I + 1280\,M$-equivalentes.
    \item \textbf{Jacobiana} (dupl.\ $4M+4S$, suma $12M+4S$, $+1$ inversión final):
          $\approx 3900\,M$-equivalentes.
  \end{itemize}
  \vspace{0.5ex}
  \begin{columns}[t]
    \begin{column}{0.5\textwidth}
      \begin{block}{Con el coste de libro ($I\approx80M$)}
        Afín $\approx 32\,000\,M$ $\Rightarrow$ razón $\approx$ \alert{8$\times$}
      \end{block}
    \end{column}
    \begin{column}{0.5\textwidth}
      \begin{block}{Con el coste real de GMP ($I\approx14M$)}
        Afín $\approx 6\,600\,M$ $\Rightarrow$ razón $\approx$ \alert{1{,}7$\times$}
      \end{block}
    \end{column}
  \end{columns}
  \vspace{0.5ex}
  Las Jacobianas no eliminan coste: \alert{intercambian} 1 inversión por $\sim$10
  multiplicaciones extra por paso. El intercambio solo es muy rentable si $I$ es cara, y
  NTL/GMP la hacen barata ($I/M$ real $\sim$10--20). Por eso el ahorro \textbf{crece} con el
  tamaño del cuerpo (1{,}72$\times$ en 256 bits $\to$ $\sim$2$\times$ en P-384).
\end{frame}

\begin{frame}[noframenumbering]{Respaldo - de la ecuación general a la forma corta}
  \scriptsize
  Partimos de la \alert{ecuación general de Weierstrass} (char$(K)\neq 2,3$):
  \[ y^2 + a_1 xy + a_3 y = x^3 + a_2 x^2 + a_4 x + a_6. \]

  \textbf{Paso 1 — completar el cuadrado en $y$} (usa char$\neq 2$). El miembro izquierdo es
  $y^2 + (a_1 x + a_3)\,y$; con $y \mapsto y - \tfrac{1}{2}(a_1 x + a_3)$ se anula el término
  lineal en $y$ y queda
  \[ y^2 = x^3 + \tfrac{b_2}{4}x^2 + \tfrac{b_4}{2}x + \tfrac{b_6}{4},
     \qquad
     b_2 = a_1^2 + 4a_2,\;\; b_4 = 2a_4 + a_1 a_3,\;\; b_6 = a_3^2 + 4a_6. \]

  \textbf{Paso 2 — eliminar el término en $x^2$} (usa char$\neq 3$). Para el cúbico
  $x^3 + p x^2 + qx + r$ (aquí $p=\tfrac{b_2}{4},\,q=\tfrac{b_4}{2},\,r=\tfrac{b_6}{4}$), la
  traslación $x \mapsto x - \tfrac{p}{3} = x - \tfrac{b_2}{12}$ lo \emph{deprime}:
  \[ \boxed{\,y^2 = x^3 + a x + b\,},\qquad
     a = \tfrac{b_4}{2} - \tfrac{b_2^2}{48},\qquad
     b = \tfrac{b_6}{4} - \tfrac{b_2 b_4}{24} + \tfrac{b_2^3}{864}. \]
  Las divisiones por $2$ (paso 1) y por $3$ (paso 2) son exactamente la razón de exigir
  char$(K)\neq 2,3$. El discriminante resulta $\Delta = -16(4a^3 + 27b^2)$.
\end{frame}

\begin{frame}[noframenumbering]{Respaldo - corrección de la verificación ECDSA}
  \footnotesize
  Si la firma es honesta, $s = k^{-1}(e + d\,r)\bmod n$, luego
  \[ k = s^{-1}(e + d\,r) = \underbrace{s^{-1}e}_{u_1} + \underbrace{s^{-1}r}_{u_2}\,d
       = u_1 + u_2\,d \pmod n. \]
  Por tanto $R' = u_1 G + u_2 Q = u_1 G + u_2\,d\,G = (u_1 + u_2 d)\,G = k\,G = R$, de donde
  $x_{R'}\bmod n = x_R \bmod n = r$ y la firma se acepta. \qed
\end{frame}

\begin{frame}[noframenumbering]{Respaldo - CRT en el descifrado RSA}
  \footnotesize
  Se calcula $d_p=d\bmod(p-1)$, $d_q=d\bmod(q-1)$, $q_{inv}=q^{-1}\bmod p$. Recombinación de
  Garner: $m_1=c^{d_p}\bmod p$, $m_2=c^{d_q}\bmod q$, $h=q_{inv}(m_1-m_2)\bmod p$,
  $m=m_2+hq$. Acelera $\approx 3.5\times$ (dos exponenciaciones a la mitad de bits).
\end{frame}
